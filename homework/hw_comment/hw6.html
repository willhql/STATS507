from sklearn.model_selection import KFold


def most_confusing_digit(X_train, y_train, X_test, y_test):
    clf = GaussianNB()
    clf.fit(X_train, y_train)
    cf_matrix = confusion_matrix(y_test, clf.predict(X_test))
    N_n = cf_matrix.sum(axis=1)
    for i in range(len(cf_matrix[0])):
        cf_matrix[i][i] = 0
    m_n = cf_matrix.sum(axis=1)
    r = m_n / N_n
    max_r = max(m_n / N_n)
    for idx, i in enumerate(r):
        if i == max_r:
            return idx


def confusion_candidates(digits):
    data = digits['data']
    target = digits['target']
    confusing_digits = []
    kf = KFold(n_splits=18, shuffle=False)
    for train_idx, test_idx in kf.split(data):
        X_train, y_train, X_test, y_test = data[train_idx], target[train_idx],\
            data[test_idx], target[test_idx]
        confusing_digits.append(most_confusing_digit(X_train, y_train, X_test, y_test))
    return confusing_digits


def majority_voting(candidates):
    counts = [0 for i in range(10)]
    for c in candidates:
        counts[c] = candidates.count(c)
    return max(counts)
